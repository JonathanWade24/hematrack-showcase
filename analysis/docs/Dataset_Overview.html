# Function to get variable information
get_var_info <- function(df) {
  map_dfr(names(df), ~{
    example_vals <- if(is.numeric(df[[.x]])) {
      paste(range(df[[.x]], na.rm = TRUE), collapse = " to ")
    } else {
      # Truncate string values to first 50 characters
      vals <- head(unique(df[[.x]]), 3)
      vals <- substr(vals, 1, 50)
      paste(vals, collapse = ", ")
    }
    
    tibble(
      Variable_Name = .x,
      Data_Type = class(df[[.x]])[1],
      N_Missing = sum(is.na(df[[.x]])),
      N_Unique = n_distinct(df[[.x]], na.rm = TRUE),
      Example_Values = example_vals
    )
  })
}

# Clean admissions data and count VOE/ACS
admissions <- clean_data_list$"IN495734_SCD_IP_Admissions_Baseline.txt" %>%
  mutate(
    has_voe = if_any(
      starts_with("admit_dx_cd_") | starts_with("final_dx_cd_"),
      ~ grepl("^D57\\.0[0-9]", .)
    ),
    has_acs = if_any(
      starts_with("admit_dx_cd_") | starts_with("final_dx_cd_"),
      ~ grepl("^D57\\.01", .)
    )
  )

# Summary of unique patients
patient_summary <- admissions %>%
  group_by(patient_mrn) %>%
  summarise(
    has_any_voe = any(has_voe & !has_acs, na.rm = TRUE),
    has_any_acs = any(has_acs, na.rm = TRUE),
    has_other = any(!has_voe & !has_acs, na.rm = TRUE)
  ) %>%
  summarise(
    Unique_VOE_Patients = sum(has_any_voe),
    Unique_ACS_Patients = sum(has_any_acs),
    Unique_Other_Patients = sum(has_other),
    Total_Unique_Patients = n()
  )

# Summary of total admissions
admission_summary <- admissions %>%
  summarise(
    VOE_Admissions = sum(has_voe & !has_acs, na.rm = TRUE),
    ACS_Admissions = sum(has_acs, na.rm = TRUE),
    Other_Admissions = sum(!has_voe & !has_acs, na.rm = TRUE),

# Function to analyze missing data
analyze_missing <- function(df) {
  map_dfr(names(df), ~{
    tibble(
      variable = .x,
      n_missing = sum(is.na(df[[.x]])),
      pct_missing = (n_missing / nrow(df)) * 100
    )
  }) %>%
    arrange(desc(pct_missing))
}

# Analyze missing data for each file
missing_data <- map(clean_data_list, analyze_missing)
